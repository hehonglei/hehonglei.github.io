    <script>  
        $(function(){  
            $('#button').click(function(){  
                var pid=$('#select').val();  
               // alert(pid);  
                var start=$('.start').val();  
                var over=$('.over').val();  
      
      
                $.getJSON("b.php",{pid:pid,start:start,over:over},function(msg){  
      
      
                    var str='';  
                    for(var i=0;i<msg['arr'].length;i++)  
                    {  
                        str=str+"<tr><td>"+msg['arr'][i].name+"</td></tr>";  
                    }  
                    $('div').html(str);  
                })  
            })  
        })  
      
    </script>
